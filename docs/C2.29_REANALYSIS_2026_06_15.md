## 3. 目前線上 view 定義（從 repo migration 推導）

### 3.1 `staff_accessible_markets`（依 `030_fix_data_isolation.sql` line 85-108 + `20240220_staff_system_simple.sql` line 178-199 推導）

```sql
CREATE VIEW staff_accessible_markets AS
-- 1. 員工可以查看老闆的市集
SELECT m.*,
       sr.owner_id AS relationship_owner_id,
       sr.permissions,
       'staff' as access_type
FROM markets m
JOIN market_members mm ON mm.market_id = m.id
JOIN staff_relationships sr ON sr.owner_id = mm.user_id
WHERE sr.staff_id = auth.uid()
  AND sr.status = 'active'

UNION ALL

-- 2. 老闆可以查看自己的市集
SELECT m.*,
       mm.user_id as relationship_owner_id,
       '{"can_view": true, "can_edit": true}'::jsonb as permissions,
       'owner' as access_type
FROM markets m
JOIN market_members mm ON mm.market_id = m.id
WHERE mm.user_id = auth.uid();
```

CREATE VIEW staff_accessible_events AS
-- 1. 員工可以查看老闆市集的事件
SELECT e.*,
       sr.owner_id AS relationship_owner_id,
       sr.permissions,
       'staff' as access_type
FROM events e
JOIN market_members mm ON mm.market_id = e.market_id
JOIN staff_relationships sr ON sr.owner_id = mm.user_id
WHERE sr.staff_id = auth.uid()
  AND sr.status = 'active'

       sr.owner_id AS relationship_owner_id,
       sr.permissions,
       'staff' as access_type

JOIN staff_relationships sr ON sr.owner_id = e.actor_id
WHERE sr.staff_id = auth.uid()
  AND sr.status = 'active'
  AND e.market_id IS NULL

UNION ALL
-- 3. 老闆可以查看自己的所有事件
SELECT e.*,
       e.actor_id as relationship_owner_id,
       '{"can_view": true, "can_edit": true}'::jsonb as permissions,

       mm.user_id as relationship_owner_id,
       '{"can_view": true, "can_edit": true}'::jsonb as permissions,

## 6. Staff payload 敏感欄位外洩風險分析

### 6.1 風險矩陣

| 攻擊向量 | 攻擊者可獲得的資料 | 嚴重度 | 既有緩解 |
|---|---|---|---|
| 員工 DevTools → `supabase.from('staff_accessible_markets').select('*')` | `booth_cost` / `commission_rate` / `total_profit` / `deposit` / `table_rental` / `chair_rental` / `umbrella_rental` / `tablecloth_rental` | 🟠 P0 | ❌ 無 |
| 員工 DevTools → `supabase.from('staff_accessible_products').select('*')` | `cost` 欄位 | 🟠 P0 | ❌ 無 |
| 員工 DevTools → `supabase.from('staff_accessible_events').select('*')` | `payload` 內 `cost` / `costAtTimeOfSale` / `supplierInfo` / `totalCost` / `profitMargin` / `grossMargin` / `totalProfit` / `boothCost` / `commissionRate` / `netProfit` / `deposit` / `tableRental` 等 | 🟠 P0 | ❌ 無 |
| 員工 DevTools → `supabase.from('markets').select('*')`（透過 RLS） | 同上（markets 全部欄位） | 🟠 P0 | ❌ 無 |
| 員工 DevTools → `supabase.from('products').select('*')`（透過 RLS） | `cost` | 🟠 P0 | ❌ 無 |
| 員工 DevTools → `supabase.from('events').select('*')`（透過 RLS） | `payload` 全部 | 🟠 P0 | ❌ 無 |

### 6.2 為什麼這是 fail-open？

C2.28 + C2.30C/D 已把**前端 UI** 與 **IndexedDB 寫入路徑** 都設為 fail-closed：
- ✅ UI 渲染前過 PermissionGate（C2.30C）
- ✅ IndexedDB 寫入前過 PermissionGate（C2.30D）
- ✅ Conflict 解決脫敏（C2.31）
- ✅ useSync 期間 infoLevel=0（C2.28）

**但**員工如果直接打 DevTools，用 Supabase JS client 查 `staff_accessible_*` view 或底表（透過 RLS），**完全繞過前端所有脫敏邏輯**。Supabase 端沒有任何脫敏：

```sql
-- 員工在 DevTools 跑這行就拿到 cost
const { data } = await supabase
  .from('staff_accessible_products')
  .select('cost, name, price')
// data[0].cost = 100
```

### 7.1 草稿 A：重建三個 view 用「列舉欄位」取代 `*`

```sql

-- 版本：039 (建議)
-- 日期：2026-06-15 (草稿，未執行)

-- 7.1.1 staff_accessible_markets

DROP VIEW IF EXISTS staff_accessible_markets CASCADE;

CREATE VIEW staff_accessible_markets AS

-- 1. 員工：列舉白名單欄位（排除敏感）
SELECT
  m.id,

  m.name,

  m.operating_start_time,

  m.table_free,
  m.chair_free,
  m.umbrella_free,
  m.tablecloth_free,
  -- 員工可見的統計（不含 profit）
  m.total_revenue,
  m.total_interactions,
  m.total_deals,
  -- ❌ 排除敏感：
  -- m.registration_fee,    -- 老闆可見
  -- m.booth_cost,          -- ❌
  -- m.deposit,             -- ❌
  -- m.table_rental,        -- ❌

  -- m.commission_rate,     -- ❌

-- 2. 老闆：保留全部欄位
SELECT

  '{"can_view": true, "can_edit": true}'::jsonb AS permissions,

FROM markets m
JOIN market_members mm ON mm.market_id = m.id

-- 7.1.2 staff_accessible_products

DROP VIEW IF EXISTS staff_accessible_products CASCADE;

CREATE VIEW staff_accessible_products AS

-- 1. 員工：列舉白名單欄位（排除 cost）
SELECT

  p.owner_id,

-- 2. 老闆：保留全部

  AND p.is_active = TRUE;

-- 7.1.3 staff_accessible_events

DROP VIEW IF EXISTS staff_accessible_events CASCADE;

CREATE VIEW staff_accessible_events AS

-- 1. 員工：白名單欄位 + JSONB payload 過濾
-- 注意：events 表本身有 id/type/actor_id/market_id/timestamp/metadata/created_at
-- 員工路徑保留 id/type/actor_id/market_id/timestamp/metadata/created_at
-- 但 payload 需過濾（用 JSONB 操作移除敏感鍵）
SELECT
  e.id,
  e.type,
  e.actor_id,
  e.market_id,
  e.timestamp,
  e.metadata,
  e.created_at,
  -- payload 過濾：員工路徑移除敏感 key
  CASE
    WHEN e.type IN ('market_created', 'market_updated')
      -- 移除成本/利潤/供應商等
      THEN e.payload - ARRAY[
        'boothCost', 'deposit', 'tableRental', 'chairRental',
        'umbrellaRental', 'tableclothRental', 'commissionRate',
        'registrationFee'
      ]
    WHEN e.type IN ('product_created', 'product_updated')
      THEN e.payload - ARRAY['cost', 'supplierInfo', 'profitMargin', 'grossMargin']
    WHEN e.type = 'deal_closed'
      THEN e.payload - ARRAY[
        'cost', 'costAtTimeOfSale', 'supplierInfo',
        'profitMargin', 'grossMargin', 'profit'
      ]
    WHEN e.type IN ('cost_added', 'cost_updated', 'cost_deleted', 'inventory_cost_updated')
      -- 員工完全不可見成本事件
      THEN NULL::jsonb
    ELSE e.payload
  END AS payload,
  sr.owner_id AS relationship_owner_id,
  sr.permissions,
  'staff' AS access_type
FROM events e
JOIN market_members mm ON mm.market_id = e.market_id
JOIN staff_relationships sr ON sr.owner_id = mm.user_id
WHERE sr.staff_id = auth.uid()
  AND sr.status = 'active'

UNION ALL

-- 2. 員工全局事件（market_id IS NULL）：同樣過濾
SELECT
  e.id,
  e.type,
  e.actor_id,
  e.market_id,
  e.timestamp,
  e.metadata,
  e.created_at,
  CASE
    WHEN e.type IN ('product_created', 'product_updated')
      THEN e.payload - ARRAY['cost', 'supplierInfo', 'profitMargin', 'grossMargin']
    WHEN e.type IN ('cost_added', 'cost_updated', 'cost_deleted', 'inventory_cost_updated')
      THEN NULL::jsonb
    ELSE e.payload
  END AS payload,
  sr.owner_id AS relationship_owner_id,
  sr.permissions,
  'staff' AS access_type
FROM events e
JOIN staff_relationships sr ON sr.owner_id = e.actor_id
WHERE sr.staff_id = auth.uid()
  AND sr.status = 'active'
  AND e.market_id IS NULL

UNION ALL

-- 3. 老闆：保留全部
SELECT
  e.*,
  e.actor_id AS relationship_owner_id,
  '{"can_view": true, "can_edit": true}'::jsonb AS permissions,
  'owner' AS access_type
FROM events e
WHERE e.actor_id = auth.uid()

UNION ALL

-- 4. 老闆：保留全部（市集成員事件）
SELECT
  e.*,
  mm.user_id AS relationship_owner_id,
  '{"can_view": true, "can_edit": true}'::jsonb AS permissions,
  'owner' AS access_type
FROM events e
JOIN market_members mm ON mm.market_id = e.market_id
WHERE mm.user_id = auth.uid()
  AND e.actor_id != auth.uid();
```

### 7.2 草稿 B：額外 RLS 政策（保護底表直接查詢）

```sql
-- 額外的 markets/products/events RLS
-- 對底表 markets/products/events 加上「staff 不可 SELECT 敏感欄位」
-- 注意：PostgreSQL RLS 是 row-level，無法控制 column-level
-- 解法：建立「column-restricted view」給 staff，把底表的 SELECT policy 改為只允許 owner

-- 7.2.1: 禁止 staff 直接 SELECT 底表（僅 owner 可）
DROP POLICY IF EXISTS "用戶可以查看自己是成員的市集" ON markets;
DROP POLICY IF EXISTS "Users can view own and team products" ON products;
DROP POLICY IF EXISTS "用戶可以查看自己的事件和市集事件" ON events;

-- 新政策：底表只允許 owner 直接讀
CREATE POLICY "markets_owner_select_only" ON markets
FOR SELECT TO authenticated
USING (owner_id = auth.uid());

CREATE POLICY "products_owner_select_only" ON products
FOR SELECT TO authenticated
USING (owner_id = auth.uid());

CREATE POLICY "events_owner_select_only" ON events
FOR SELECT TO authenticated
USING (actor_id = auth.uid());

-- 員工只能透過 view 讀
-- view 內部用 SECURITY DEFINER function 繞過 RLS
```

**注意**：草稿 B 會破壞現有前端的 `lib/supabase/markets.ts` / `products.ts` / events 相關查詢（這些 query 不走 view）。**需要在套用前先遷移前端到 view 為唯一入口**。
